<script>
    class AFD{

constructor(alphabet, q0, F, transitionTable){
    this.alphabet = alphabet;
    this.q0 = q0;
    this.F = F;
    this.transitions = transitionTable;
    this.backtrack = [];
}

getTransitions(){
    return JSON.parse(JSON.stringify(this.transitions));
}

getF(){
    return this.F;
}

getVisitedNodes(){
    return this.backtrack;
}

// Returns true if automata accepts the word
accepts(word){
    // restart the backtrack queue
    this.backtrack = [];
    // Verify that every symbol belongs to the alphabet
    for (var char of word){
        if (!this.alphabet.has(char)) return false;
    }
    let q = this.q0;
    q = this.transitionFunc(q, word);
    if (this.F.has(q)) { 
        return true;
    }
    return false;
}

transitionFunc(q, word){
    // Base case
    if (word.length === 0){
        this.backtrack.push(q);
        return q;
    }
    let w = word.slice(0, -1);
    let o = word.slice(-1);
    // Get the state that will result from w
    q = this.transitionFunc(q, w);
    q = this.transitions[q][o];
    this.backtrack.push(q);
    return q
}
}

<script>
    class AFN{

constructor(alphabet, q0, F, transitionTable){
    this.alphabet = alphabet;
    this.q0 = q0;
    this.F = F;
    this.transitions = transitionTable;
    this.backtrack = [];
}

getTransitions(){
    return this.transitions;
}

getF(){
    return this.F;
}

getVisitedNodes(){
    return this.backtrack;
}

// Returns true if automata accepts the word
accepts(word){
    // Reset the backtrack
    this.backtrack = [];
    // Keep track of every state
    let endStates = [];
    // Verify that every symbol belongs to the alphabet
    for (var char of word){
        if (!this.alphabet.has(char)) return false;
    }
    let A = new Set();
    // Add initial state to set
    A.add(this.q0);
    // Call deltaTestate on the word
    this._deltaTestate(A, word);
    endStates.push(A);
    for (var state of A){
        if (this.F.has(state)) {
            endStates.push(true);
            return endStates;
        }
    }
    endStates.push(false);
    return endStates;
}

_deltaTestate(A, word){
    // Base case
    if (word.length === 0){
        // See if p can be accessed from q e A through an epsilon transition
        for (var q of A) {
            let P = this.transitions[q][''];
            if (P) {
                // There exists an epsilon transition from q and it reaches p:
                for (var p of P){
                    // Add to set reachable states p
                    A.add(p);
                }
            }
        }
        this.backtrack.push(new Set(A));
        return;
    }
    let w = word.slice(0, -1);
    let a = word.slice(-1);
    // Recursively fill the states
    this._deltaTestate(A, w);
    this._delta(A, a);
    
    return;
}

_delta(q, a){
    if (a.length !== 1){
        return;
    }
    // Set the states
    let newStates = new Set();
    for (var state of q){
        // Union operator
        newStates = new Set([...this.transitions[state][a], ...newStates]);
    }
    q.clear();
    for(var state of newStates){
        q.add(state);
    }

    // See if p can be accessed from qs through an epsilon transition
    for (var qs of q) {
        let P = this.transitions[qs][''];
        if (P) {
            // There exists an epsilon transition from qs and it reaches p:
            for (var p of P){
                // Add to q reachable states p
                q.add(p);
            }
        }
    }
    this.backtrack.push(new Set(q));
    return;
}
}

    const COLOR_DEFAULT = '#46961b';
    const SELECTED_COLOR = '#f4a742';
    const EDGE_COLOR = '#777777';
    const MODES = ['NORMAL', 'IDENTIFIER', 'NUMBER'];
    document.querySelector('.title').innerHTML = "Práctica Lenguajes Regulares";
    
    let curr_mode;
    let automaton;
    let automatonTransitions;
    let finalStates;
    let nodes;
    let edges;
    let network;

    // create a network
    var container = document.getElementById('mynetwork');

    // Function used to initialized automaton
    const initialize_automaton = (isAFN, mode, alphabet, q, F, transitions) => {
        automaton = (isAFN)? new AFN(alphabet, q, F, transitions):new AFD(alphabet, q, F, transitions);
        automatonTransitions = automaton.getTransitions();
        finalStates = automaton.getF();

        nodes = new vis.DataSet([]);
        edges = new vis.DataSet([]);

        curr_mode = MODES[mode];

        if (automaton instanceof AFD){
            Object.keys(automatonTransitions).forEach((key) => {
                let node = {};
                node.id = key;
                node.shape = 'circle';
                node.label = key;
                node.color = {background: COLOR_DEFAULT, border: '#000000'};
                node.borderWidth = (finalStates.has(key))? 3: 0;
                nodes.add(node);
                // Add edges
                edgeAux = [];
                Object.keys(automatonTransitions[key]).forEach(symbol=>{
                    // Check if 'to' is already on the edge array
                    let repeated = false;
                    for (var i = 0; i < edgeAux.length; i++) {
                        if (edgeAux[i].to === automatonTransitions[key][symbol]){
                            // Found repeated destination
                            repeated = true;
                            // Change label of previous node
                            edgeAux[i].label = edgeAux[i].label + ", " + symbol;
                        }
                    }
                    if(!repeated) {
                        let edge = {};
                        edge.from = key;
                        edge.to = automatonTransitions[key][symbol];
                        edge.label = symbol;
                        edge.arrows = 'to';
                        edge.color = {color: EDGE_COLOR};
                        edgeAux.push(edge);
                    }
                });
                // Add edges of current node
                for(var i = 0; i < edgeAux.length; i++){
                    edges.add(edgeAux[i]);
                }
            });
        }else if (automaton instanceof AFN){
            Object.keys(automatonTransitions).forEach((key) => {
                let node = {};
                node.id = key;
                node.shape = 'circle';
                node.label = key;
                node.color = {background: COLOR_DEFAULT, border: '#000000'};
                node.borderWidth = (finalStates.has(key))? 3: 0;
                nodes.add(node);
                // Add edges
                edgeAux = [];
                Object.keys(automatonTransitions[key]).forEach(symbol=>{
                    // Chec sets inside each symbol
                    for (let toState of automatonTransitions[key][symbol]){
                        // Check if 'to' is already on the edge array
                        let repeated = false;
                        for (var i = 0; i < edgeAux.length; i++) {
                            if (edgeAux[i].to === toState){
                                // Found repeated destination
                                repeated = true;
                                // Change label of previous node only if length is less than 3
                                let letters = edgeAux[i].label.split(', ');
                                if (letters.length < 2){
                                    edgeAux[i].label = edgeAux[i].label + ", " + ((symbol === '')?'ε':symbol);
                                }else{
                                    edgeAux[i].label = "";
                                    for (var letter = 0; letter < letters.length-1; letter++){
                                        edgeAux[i].label += letters[letter] + ", ";
                                    }
                                    edgeAux[i].label += '..' + symbol;
                                    
                                }
                            }
                        }
                        if(!repeated) {
                            let edge = {};
                            edge.from = key;
                            edge.to = toState;
                            edge.label = (symbol === '')?'ε':symbol;
                            edge.arrows = 'to';
                            edge.color = {color: EDGE_COLOR};
                            edgeAux.push(edge);
                        }
                    }
                });
                // Add edges of current node
                for(var i = 0; i < edgeAux.length; i++){
                    edges.add(edgeAux[i]);
                }
            });
        }

        // provide the data in the vis format
        var data = {
            nodes: nodes,
            edges: edges
        };
        var options = {
            layout: {
                randomSeed: 548308,
                improvedLayout: true
            }
        };

        // initialize your network!
        network = new vis.Network(container, data, options);
    }

    const startAnimation = (backtrack, speed, word, callback) => {
        var promise = Promise.resolve();
        const n = backtrack.length;
        backtrack.forEach((state, i) => {
            promise = promise.then(() => {
                if (i >= 1) {
                    document.querySelector('#symbol').innerHTML = word.charAt(0);
                    word = word.slice(1);
                }
                // Unselect previous colored state in backtrack if exists
                if (i >= 1) {
                    if (automaton instanceof AFD){
                        // repaint node
                        const prev_node = backtrack[i-1];
                        nodes.update({
                            id: prev_node,
                            color: {background: COLOR_DEFAULT, border: '#000000'}
                        });
                        // repaint edge
                        if (i >= 2){
                            const prev_prev_node = backtrack[i-2];
                            Object.keys(edges._data).forEach((edge) => {
                                if (edges._data[edge].from == prev_prev_node && edges._data[edge].to == prev_node) {
                                    edges.update({
                                        id: edge,
                                        color: {color: EDGE_COLOR}
                                    });
                                }
                            });
                        }
                    }else if (automaton instanceof AFN){
                        // backtrack is a set of states and we need to unpaint all of them
                        for (const s of backtrack[i-1]) {
                            nodes.update({
                                id: s,
                                color: {background: COLOR_DEFAULT, border: '#000000'}
                            });
                            // repaint edge
                            if (i >= 2){
                                const prev_prev = backtrack[i-2];
                                for (var prev_prev_node of prev_prev){
                                    Object.keys(edges._data).forEach((edge) => {
                                        if (edges._data[edge].from == prev_prev_node && edges._data[edge].to == s) {
                                            edges.update({
                                                id: edge,
                                                color: {color: EDGE_COLOR}
                                            });
                                        }
                                    });
                                }
                            }
                        }
                    }

                if (automaton instanceof AFD) {
                    
                    if (i >= 1){

                        Object.keys(edges._data).forEach((edge) => {
                            if (edges._data[edge].from == prev_node && edges._data[edge].to == state) {
                                edges.update({
                                    id: edge,
                                    color: {color: SELECTED_COLOR}
                                });
                            }
                        });
                    }
                    nodes.update({
                        id: state,
                        color: SELECTED_COLOR
                    });
                }else if (automaton instanceof AFN) {
                    // state is a set of states and we need to paint all of them
                    for (const s of state) {
                        if (i >= 1){
                            const prev = backtrack[i-1];
                            for (var prev_node of prev) {
                                Object.keys(edges._data).forEach((edge) => {
                                    if (edges._data[edge].from == prev_node && edges._data[edge].to == s) {
                                        edges.update({
                                            id: edge,
                                            color: {color: SELECTED_COLOR}
                                        });
                                    }
                                });
                            }
                        }
                        nodes.update({
                            id: s,
                            color: SELECTED_COLOR
                        });
                    }
                }
                
                return new Promise((resolve) => {
                    if (i < n){
                        setTimeout(resolve, speed);
                    }else{
                        resolve();
                    }
                });
            });
        });

        promise.then(() => {
            document.querySelector('#symbol').innerHTML = "";
            if (automaton instanceof AFD){
                const prev = backtrack[backtrack.length-1];
                const prev_prev = backtrack[backtrack.length-2];
                nodes.update({
                    id: prev,
                    color: {background: COLOR_DEFAULT, border: '#000000'}
                });
                // repaint edge
                if (prev_prev){
                    Object.keys(edges._data).forEach((edge) => {
                        if (edges._data[edge].from == prev_prev && edges._data[edge].to == prev) {
                            edges.update({
                                id: edge,
                                color: {color: EDGE_COLOR}
                            });
                        }

                }
            }else if (automaton instanceof AFN){
                // backtrack is a set of states and we need to unpaint all of them
                const prev = backtrack[backtrack.length-1];
                for (const s of prev) {

                                if (edges._data[edge].from == prev_prev_node && edges._data[edge].to == s) {
                                    edges.update({
                                        id: edge,
                                        color: {color: EDGE_COLOR}
                                    });
                                }
                            });
                        }
                    }
                }
            }
            callback();
        });

    available = true;

        if (curr_mode === MODES[2]){

        document.querySelector("#result").className = "";
        document.querySelector("#result h3").innerHTML = "";
        // Input values
        let word = document.querySelector('#wordInput').value;
        const speed = speedRange.value*10;

        // Get results
        results = automaton.accepts(word);
        if (automaton instanceof AFN){
            endStates = results[0];
            valid = results[1];
        }else{
            valid = results;
        }
        backtrack = automaton.getVisitedNodes();

        if (backtrack.length === 0 || endStates.size == 0){

            initialize_automaton(true, 2, alphabet_2, 'q0', F_2, transitions_2);

            if (curr_mode == MODES[2] && backtrack.length !== 0){

                }else if (endStates.has('q3')){

                case MODES[0]:
